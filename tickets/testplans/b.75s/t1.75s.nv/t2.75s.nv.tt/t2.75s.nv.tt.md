# cabi-5-double-close-noop

Pins Epic 2 AC: `ad_close` with an unknown or already-deleted handle is a
success no-op — the desired post-condition (no active session) is already met.
The sequence `ad_open` → `ad_close(handle)` → `ad_close(handle)` must produce
three success envelopes; closing a never-minted handle must also succeed.

## Setup

No DB state needed for the C-ABI registry check itself. A SQLite file is
created at `/tmp/probe5_state.db` by `ad_open` (using `create_if_missing:
true`). The db-reset fixture does not touch `/tmp/`, so this file is isolated.

All four operations (open + three closes) run in a single C probe process so
the in-process handle registry persists across calls.

## Steps

1. Assert `gcc` is available in PATH; fail with a clear message if not.
2. Build `.so` to `/tmp/ad_cabi_test.so` if not already present.
3. Write a C probe that, in a single `dlopen` session:
   a. Calls `ad_open({"store_path":"/tmp/probe5_state.db","create_if_missing":true})`.
   b. Extracts the `handle` string from the returned JSON.
   c. Calls `ad_close({"handle":"<handle>"})` — first close.
   d. Calls `ad_close({"handle":"<handle>"})` again — second close (no-op).
   e. Calls `ad_close({"handle":"00000000000000000000000000000001"})` — unminted handle close.
   f. Asserts none of the three close results contain `"err_name"`.
   g. Prints `all_ok` and exits 0 if all three closes succeeded.
4. Compile with `gcc /tmp/probe5.c -ldl -o /tmp/probe5`.
5. Run `/tmp/probe5 "$SO"` and assert stdout contains `all_ok`.

## Pass criteria

- `/tmp/probe5` exits 0.
- `ad_open` returns a handle (no `err_name`).
- All three `ad_close` calls return envelopes without `err_name`.
- Stdout contains the line `all_ok`.
- Script exits 0; final line matches `cabi-5 ok: ...`.

## Script

```bash
set -euo pipefail

if ! command -v gcc >/dev/null 2>&1; then
    echo "FAIL: gcc not found — add build-essential to test/Dockerfile"
    exit 1
fi

SO="/tmp/ad_cabi_test.so"
if [[ ! -f "$SO" ]]; then
    WORK=$(mktemp -d /tmp/cabi5_XXXXXX)
    cp -a /work/source/. "$WORK/"
    ( cd "$WORK" && mkdir -p dist && CGO_ENABLED=1 go build -buildmode=c-shared -o "$SO" ./pkg/cabi )
fi

cat > /tmp/probe5.c << 'CEOF'
#include <dlfcn.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef char *(*ad_fn_t)(const char *);
typedef void (*free_fn_t)(char *);

/* Minimal handle extraction: find "handle":"<token>" in a JSON string */
static int extract_handle(const char *json, char *out, int outlen) {
    const char *key = "\"handle\":\"";
    const char *p = strstr(json, key);
    if (!p) return 0;
    p += strlen(key);
    const char *end = strchr(p, '"');
    if (!end) return 0;
    int len = (int)(end - p);
    if (len <= 0 || len >= outlen) return 0;
    memcpy(out, p, len);
    out[len] = '\0';
    return 1;
}

/* Returns 1 if the JSON string contains an "err_name" key */
static int has_err_name(const char *json) {
    return strstr(json, "\"err_name\"") != NULL;
}

int main(int argc, char **argv) {
    if (argc < 2) { fputs("usage: probe5 <so_path>\n", stderr); return 1; }

    void *lib = dlopen(argv[1], RTLD_NOW);
    if (!lib) { fprintf(stderr, "dlopen: %s\n", dlerror()); return 1; }

    ad_fn_t fn_open  = (ad_fn_t)dlsym(lib, "ad_open");
    ad_fn_t fn_close = (ad_fn_t)dlsym(lib, "ad_close");
    free_fn_t freefn = (free_fn_t)dlsym(lib, "ad_free_cstring");
    if (!fn_open || !fn_close || !freefn) {
        fprintf(stderr, "dlsym failed: %s\n", dlerror());
        dlclose(lib);
        return 1;
    }

    /* 1. Open — mint a handle */
    char *open_res = fn_open(
        "{\"store_path\":\"/tmp/probe5_state.db\",\"create_if_missing\":true}"
    );
    char handle[256] = {0};
    int have_handle = extract_handle(open_res, handle, (int)sizeof(handle));
    printf("open: %s\n", open_res);
    if (has_err_name(open_res)) {
        fputs("FAIL: ad_open returned an error\n", stdout);
        freefn(open_res);
        dlclose(lib);
        return 1;
    }
    freefn(open_res);

    if (!have_handle) {
        fputs("FAIL: ad_open did not return a handle field\n", stdout);
        dlclose(lib);
        return 1;
    }

    /* 2. First close — removes handle from registry */
    char close_params[512];
    snprintf(close_params, sizeof(close_params), "{\"handle\":\"%s\"}", handle);
    char *close1 = fn_close(close_params);
    printf("close1: %s\n", close1);
    int err1 = has_err_name(close1);
    freefn(close1);

    /* 3. Second close — same handle now unknown, must be no-op success */
    char *close2 = fn_close(close_params);
    printf("close2: %s\n", close2);
    int err2 = has_err_name(close2);
    freefn(close2);

    /* 4. Close a handle that was never minted */
    char *close3 = fn_close(
        "{\"handle\":\"00000000000000000000000000000001\"}"
    );
    printf("close3: %s\n", close3);
    int err3 = has_err_name(close3);
    freefn(close3);

    dlclose(lib);

    if (err1 || err2 || err3) {
        printf("FAIL: close errors — close1_err=%d close2_err=%d close3_err=%d\n",
               err1, err2, err3);
        return 1;
    }
    puts("all_ok");
    return 0;
}
CEOF
gcc /tmp/probe5.c -ldl -o /tmp/probe5

OUTPUT=$(/tmp/probe5 "$SO")

if ! printf '%s' "$OUTPUT" | grep -q '^all_ok$'; then
    echo "FAIL: probe5 did not emit 'all_ok':"
    printf '%s\n' "$OUTPUT"
    exit 1
fi

echo "cabi-5 ok: ad_close is a no-op success for unknown and double-closed handles"
```
